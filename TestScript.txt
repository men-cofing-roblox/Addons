local chars = "abcdefghijklmnopqrstuvwxyz0123456789"
function randomString(length)
    local result = ""
    for i = 1, length do
        local rand = math.random(1, #chars)
        result = result .. chars:sub(rand, rand)
    end
    return result
end
local library = loadstring(game:HttpGet("https://raw.githubusercontent.com/igorgames-htm/SplixUiLib/refs/heads/main/Main?"..randomString(16)))()
local window = library:new({textsize = 13.5,font = Enum.Font.RobotoMono,name = "Men Config Addons",color = Color3.fromRGB(110,110,110),mobile = "MenConfig"})
window.key = Enum.KeyCode.P
window.x = false
window.y = false
local thisgameui = window:page({name = "ThisGame"})
local gametab = thisgameui:section({name = "Main",side = "left",size = sizetab})

local libui = {["_Tab"] = gametab}

libui.CreateButton = function(self,data)
    local namelib = data.Name or data.name or "Button"
    local callbacklib = data.Callback or data.callback or function()end
    self["_Tab"]:button({name = namelib,callback = callbacklib})
    return
end

libui.CreateToggle = function(self,data)
    local namelib = data.Name or data.name or "Toggle"
    local callbacklib = data.Callback or data.callback or function(value)end
    local currentvalue = data.CurrentValue or data.currentvalue or false
    local setidk = self["_Tab"]:toggle({name = namelib,def = currentvalue,callback = callbacklib})
    return {["Set"] = function(...)
        setidk:set(...)
    end}
end

libui.CreateSlider = function(self,data)
    local namelib = data.Name or data.name or "Slider"
    local callbacklib = data.Callback or data.callback or function(value)end
    local range = data.Range or data.range or {0,100}
    local roundinglib = data.Rounding or data.rounding or true
    local currentvalue = data.CurrentValue or data.currentvalue or 10
    local suffix = data.Suffix or data.suffix or "Banana"
    local setidk = self["_Tab"]:slider({name = namelib,def = currentvalue, max = range[2],min = range[1],rounding = roundinglib,ticking = false,measuring = suffix,callback = callbacklib})
    return {["Set"] = function(...)
        setidk:set(...)
    end}
end

libui.CreateDropdown = function(self,data)
    local namelib = data.Name or data.name or "Dropdown"
    local callbacklib = data.Callback or data.callback or function(value)end
    local currentoption = data.CurrentOption or data.currentoption or {"Option 1"}
    local optionslib = data.Options or data.options or {"Option 1", "Option 2"}
    local multipleoptions = data.MultipleOptions or data.multipleoptions or false
    local setidk = nil
    if multipleoptions then
        setidk = self["_Tab"]:multibox({name = namelib,def = currentoption, max = 6,options = optionslib,callback = callbacklib})
    else
        setidk = self["_Tab"]:dropdown({name = namelib,def = currentoption[1],max = 6,options = optionslib,callback = function(...) callbacklib({...}) end})
    end
    return {["Set"] = function(...)
        setidk:set(...)
    end}
end
