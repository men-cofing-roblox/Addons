local addons = {}

local load = function(GameID,Library)
	local _1 = false
	local player = game.Players.LocalPlayer
local char = player.Character or player.CharacterAdded:Wait()

local ReplicatedStorage = game:GetService("ReplicatedStorage")
local pinchRemote = ReplicatedStorage.COM.Pinch.LetMeGo

function onchar()
	if char:GetAttribute("Grabbed") and _1 then
		pinchRemote:FireServer()
	end
end

char:GetAttributeChangedSignal("Grabbed"):Connect(function()
	if char:GetAttribute("Grabbed") and _1 then
		pinchRemote:FireServer()
	end
end)
end
onchar()
player.CharacterAdded:Connect(function(chara)
	char = chara
	onchar()
end)
	Library.CreateToggle(GameID,{
   ["Name"] = "anit grab",
   ["CurrentValue"] = false,
   ["Callback"] = function(Value)
	_1 = Value
   end,
})
end

addons.Load = load

return addons
