local function load(Tab)
local Players = game:GetService("Players")
local player = Players.LocalPlayer
local function if_player_is_on_round_or_another(player)
	return player:GetAttribute("EnvironmentID") ~= nil
end
local function findStringInTable(tbl, search)
    for _, value in ipairs(tbl) do
        if value == search then
            return true
        end
    end
    return false
end
local function chid_to_id(chid)
	return string.byte(chid or string.char(0))
end
local function get_every_player_in_Environment_but_without_thisteam(id,id2)
	local output = {}
	for k, v in pairs(game:GetService("Players"):GetChildren()) do
		if v:IsA("Player") and if_player_is_on_round_or_another(v) and chid_to_id(v:GetAttribute("EnvironmentID")) == id and chid_to_id(v:GetAttribute("TeamID")) ~= id2 then
			table.insert(output, v)
		end
	end
	return output
end
local function getplayerloadout(player)
    local hasit = ""
    local itemsleft = {}
    if player == game:GetService("Players").LocalPlayer then
        hasit = string.split(workspace.ViewModels.FirstPerson:GetChildren()[1].Name," - ")[2]
    else
        for _,v in pairs(workspace.ViewModels:GetChildren()) do
            local split = string.split(v.Name," - ")
            if split[1] == player.Name then
                hasit = split[2]
                break
            end
        end
    end
    for _,v in pairs(game:GetService("ReplicatedStorage").Assets.Temp.ViewModels:GetChildren()) do
        local split = string.split(v.Name," - ")
        if split[1] == player.Name then
            table.insert(itemsleft,split[2])
        end
    end
    return {has=hasit,left=itemsleft}
end
local function isneedtobefront(player)
    local loadout = getplayerloadout(player)
    local ishasriot = false
    local ishasonkatana = loadout.has == "Katana"
    local ishasonriot = loadout.has == "Riot Shield"
    for i,v in loadout.left do
		if v == "Riot Shield" then
			ishasriot = true
			break
		end
    end
    return ishasriot or (ishasonkatana or not ishasonriot)
end
